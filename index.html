\[ \frac{\partial V}{\partial t} + \frac{1}{2}\sigma^2S^2\frac{\partial^2 V}{\partial S^2} + rS\frac{\partial V}{\partial S} - rV = 0 \]
function calculateRisk(S, r, sigma) {
  let dt = 0.01;
  let V = S;
  for (let i = 0; i < 100; i++) {
    V += r * V * dt;
    V += sigma * S * Math.sqrt(dt);
  }
  return V;
}
